try:
    # Define the seed in hexadecimal string format
    seed_hex = '577BF2C26E45C460'  # Example seed

    # Send the seed to the server
    print(f"Sending seed: {seed_hex}")
    sent = sock.sendto(seed_hex.encode(), server_address)

finally:
